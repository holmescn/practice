%
% 首先，要定义环境变量
%
% export OSFONTDIR=$HOME/.fonts
%
% 其次将type-adobe.tex 复制到 $CONTEXTDIR/tex/texmf-fonts/context/user
%
% 执行
%
% context --generate
%
% 使用：
%
% \usetypescriptfile[type-adobe]
% \setupbodyfont[adobe, rm, 9pt]
